chmod [опции] [режим] файл - используется для изменения прав доступа к файлам и директориям
	u — владелец файла (user),
	g — группа (group),
	o — остальные (others),
	a — все категории (all).
	
	r — чтение (read),
	w — запись (write),
	x — выполнение (execute).
	
	+ — добавление прав,
	- — удаление прав,
	= — установка прав.

	-R — рекурсивное изменение прав (для директорий и их содержимого).
	--reference=файл — применить те же права, что и у указанного файла.

сhown [опции] [новый_владелец][:новая_группа] файл - используется  для изменения владельца и/или группы файла или директории
	-R — рекурсивное изменение владельца/группы для всех файлов и папок внутри директории.
	--reference=файл — взять владельца и группу из указанного файла.
	--silent или -f — подавить сообщения об ошибках.

chgrp [опции] [новая_группа] файл - используется для изменения группы, которой принадлежит файл или директория
